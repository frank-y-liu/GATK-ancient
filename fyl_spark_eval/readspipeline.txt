## Readspipeline performs both BQSR computation and applying BQSR, as well as markduplicates
## The output is a bam file which can be directly used for variant calling 
## 
## Remove the interval list file to run on whole genome 
## 
$GATK_HOME/gatk-launch \
    ReadsPipelineSpark \
    -I $HDFS_INPUT/WGS-G94982-NA12878.bam \
    -knownSites $SNP_INPUT/dbsnp_138.hg19.vcf.gz \
    -L Broad.human.exome.b37.interval_list \
    -R $HDFS_INPUT/human_g1k_v37.2bit \
    -O $HDFS_OUTPUT/Test_WES_ReadsPipeline.bam \
    -- \
    --sparkRunner <sparkrunner> \
    --sparkMaster <sparkmaster> \
    --executor-cores <num_cores> \
    --executor-memory <exec_mem> \
    --driver-memory <driver_mem>
